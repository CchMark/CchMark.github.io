IEnumerable 和 List 區別

IEnumerable 與 List 差異

C# 中 IEnumerable 和 List 的主要區別在於 IEnumerable 是一個介面，而 List 是一個具體的class可以new 出來實作。此外， IEnumerable 是只讀的，而 List 不是。 List 提供Add、Clear、Contains、IndexOf、RemoveAt、Count等方法，而 IEnumerable 提供了一個介面來接收集合資料，用於逐一讀取集合中的資料內容（枚舉）。

C# 如何做分頁

  //每頁顯示資料
  const int pageSize = 2;
  //頁碼 初始0 為 畫面呈現時的第一頁
  int pageNum = 0;

  //建立array 資料   
  string[] names = { "a", "b", "c", "d", "e", "f", "g", "h", "i","j","k" };

  while (pageNum * pageSize < names.Length)
  {
      //分頁  skip()略過頁面筆數 Take()取得筆數
      var query = names.Skip(pageNum * pageSize).Take(pageSize);
      Console.WriteLine("輸出第{0}頁紀錄", pageNum + 1);
      //逐一輸出內容   
      foreach (var q in query)
      {
          Console.WriteLine(q);
      }
      pageNum++;
  }

var client = new RestClient("http://example.com");

//為身分驗證器新增 帳號 密碼
//client.Authenticator = new HttpBasicAuthenticator(username, password);

var request = new RestRequest("resource/{id}", Method.POST);

//基於方法添加到 POST 或 URL 查詢字符串 
//adds to POST or URL querystring based on Method
request.AddParameter("name", "value");  

//替換request.Resource中匹配的token 
//replaces matching token in request.Resource
request.AddUrlSegment("id", "123"); 

//為對象的所有屬性添加參數
//add parameters for all properties on an object
request.AddObject(object);

//或者只是列入白名單的屬性
// or just whitelisted properties
request.AddObject(object, "PersonId", "Name", ...);

//容易新增HTTP 標頭
// easily add HTTP Headers
request.AddHeader("header", "value");

// 添加要上傳的文件（適用於兼容動詞）
// add files to upload (works with compatible verbs)
request.AddFile("file", path);

// 執行請求
// execute the request
IRestResponse response = client.Execute(request);

// 原始內容為字符串
// raw content as string
var content = response.Content;

// 或者自動反序列化結果
// 返回內容類型為<Person>，但可以通過 RestClient.AddHandler() 顯式設置；
// or automatically deserialize result
// return content type is sniffed but can be explicitly set via RestClient.AddHandler();
IRestResponse<Person> response2 = client.Execute<Person>(request);
var name = response2.Data.Name;

//或下載檔並將其保存到磁碟
// or download and save file to disk
client.DownloadData(request).SaveAs(path);

//簡單的異步支援
// easy async support
await client.ExecuteAsync(request);

//異步反序列化
// async with deserialization
var asyncHandle = client.ExecuteAsync<Person>(request, response => {
    Console.WriteLine(response.Data.Name);
});

//按需中止請求
// abort the request on demand
asyncHandle.Abort();

CSVToDataTable 指定某字串為結束位置

筆記一下NPOI 轉 DataTable 字串之中有換行符號導致提前讀取結束，指定結束位置的對應字串

        public static DataTable CsvToDataTableByNewEgg(string fileName)
        {
            DataTable dataTable = new DataTable();
            try
            {
                using (StreamReader sr = new StreamReader(fileName))
                {
                    char[] separators = new char[] { '#', '"', ' ' };
                    string[] columnheaders = sr.ReadLine().Split(',');

                    foreach (string header in columnheaders)
                    {
                        dataTable.Columns.Add(header.Trim(separators));
                    }
                    dataTable.Columns.Add("FileName", typeof(string));
                    dataTable.Columns.Add("CreDate", typeof(DateTime));
                    var index = 0;
                    while (!sr.EndOfStream)
                    {
                        if (index > 0)
                        {
                            var readRows = new List<string>();
                            var row = string.Empty;
                            var isReadEnd = false;
                            do
                            {
                                //讀CSV
                                row = sr.ReadLine();
                                readRows.Add(row);
                                if (row.Length >= 8)
                                {
                                    if (row.Substring(row.Length - 6, 6) == "Taiwan")
                                    {
                                        isReadEnd = true;
                                    }
                                }
                                //依據是否讀Tainwan為結束位置
                            } while (!isReadEnd);
                            string[] rows = string.Join("", readRows).Split(',');


                            DataRow dr = dataTable.NewRow();
                            for (int i = 0; i < columnheaders.Length; i++)
                            {
                                dr[i] = rows[i];
                            }
                            dr["FileName"] = "tt";
                            dr["CreDate"] = DateTime.Now;
                            dataTable.Rows.Add(dr);
                        }

                        index++;
                    }
                }
            }
            catch (Exception ex)
            {
                throw new Exception("Csv to DataTable: \n" + ex.Message);
            }
            return dataTable;
        }
